# useDebounceEvent

This hook generates an easy to use debounced event handler for input changes.

<Canvas of={UseDebounceEventStories.Overview} />

### Import

```js
import { useDebounceEvent } from "@vibe/core";
```

## Arguments

<FunctionArguments>
  <FunctionArgument name="options" type="Object">
    <FunctionArgument name="trim" type="boolean" description="Whether to trim the value or not." default="false" />
    <FunctionArgument
      name="onChange"
      type="(value: string) => void"
      description="Callback function to execute on changes."
      default="() => null"
    />
    <FunctionArgument
      name="delay"
      type="number"
      description="The amount of time (in ms) to delay the value's update."
      default="0"
    />
  </FunctionArgument>
</FunctionArguments>

## Returns

<FunctionArguments>
  <FunctionArgument name="result" type="Object">
    <FunctionArgument
      name="onEventChanged"
      type="(event: Event) => void"
      description={
        <>
          A wrapper around the passed <code>onChange</code> function.
        </>
      }
    />
    <FunctionArgument name="clearValue" type="() => void" description="Calls onChange with an empty string." />
  </FunctionArgument>
</FunctionArguments>
