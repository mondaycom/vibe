## Overview

Avatar is a graphical representation of a person through a profile picture, image, icon, or set of initials.

## Do’s and Don’ts

<ComponentRules
  rules={[
    {
      positive: {
        component: (
          <div className="monday-storybook-avatar_multiple">
            <Avatar className="multiple-avatar" size="large" src={person1} type="img" />
            <Avatar className="multiple-avatar" size="large" src={person2} type="img" />
            <Avatar className="multiple-avatar" size="large" src={person3} type="img" />
          </div>
        ),
        description: "Use consistent avatar sizes for common use cases to convey their purpose."
      },
      negative: {
        component: (
          <div className="monday-storybook-avatar_multiple-reverse">
            <Avatar className="multiple-reverse-avatar" size="small" src={person1} type="img" />
            <Avatar className="multiple-reverse-avatar" size="medium" src={person2} type="img" />
            <Avatar className="multiple-reverse-avatar" size="large" src={person3} type="img" />
          </div>
        ),
        description: "Avoid using a mix of avatar sizes that display together and create design imbalance."
      }
    },
    {
      positive: {
        component: <Avatar src={maleIcon} type="img" />,
        description: "Use branded generic avatars when a user has not set their avatar image."
      },
      negative: {
        component: <Avatar src={femaleIcon} type="img" />,
        description: "Don’t make assumptions and use gendered placeholder avatars."
      }
    }
  ]}
/>
