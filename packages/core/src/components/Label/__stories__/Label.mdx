## Do’s and Don’ts

<ComponentRules
  rules={[
    {
      positive: {
        component: <Label text="New" />,
        description: "Use label to indicate the status of an item, for example: “New”."
      },
      negative: {
        component: <Label text="123" />,
        description: (
          <>
            Don’t use the label component in order to indicate numbers, instead use the{" "}
            <StorybookLink page="Components/Counter">Counter.</StorybookLink>
          </>
        )
      }
    },
    {
      positive: {
        component: (
          <div className="monday-storybook-label_inline-block">
            <h5 className="monday-storybook-label_title">Categories</h5>
            <Label text="new" />
          </div>
        ),
        description: "Use label only once per item."
      },
      negative: {
        component: [
          <div className="monday-storybook-label_block">
            <h5>Categories</h5>
            <div className="monday-storybook-label_inline-block">
              <Label text="Featured" />
              <Label text="CRM" />
              <Label text="Export" />
            </div>
          </div>
        ],
        description: (
          <>
            Don’t use multiple labels for one item. Instead, use{" "}
            <StorybookLink page="Components/Chips">Chips.</StorybookLink>
          </>
        )
      }
    },
    {
      positive: {
        component: [<Label text="New" kind="line" />, <Label text="Beta" />],
        description: "Use only the UI colors above."
      },
      negative: {
        component: [
          <Label className="monday-storybook-label_bad-lable monday-storybook-label_purple" kind="line" text="Beta" />,
          <Label className="monday-storybook-label_bad-lable monday-storybook-label_pink" kind="line" text="Beta" />
        ],
        description: "Don’t use any content colors for labels. If the page is full of CTAs, use the outline state."
      }
    }
  ]}
/>
